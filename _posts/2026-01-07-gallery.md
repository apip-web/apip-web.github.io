.gallery {
  column-count: 2; /* 2 kolom */
  column-gap: 12px;

  margin-bottom: 12px;

  width: 100%;
  height: auto;
  display: block;
}
</style>

<script>
(function () {
  const CLOUD_NAME = 'dpawix5pp';
  const MAX_WIDTH = 1200;

  // Definisikan container dan range gambarnya
  const galleries = [
    { selector: '.gallery-1', start: 1, end: 6 },
    { selector: '.gallery-2', start: 7, end: 12 }
  ];

  galleries.forEach(g => {
    const gallery = document.querySelector(g.selector);
    if (!gallery) return;

    for (let i = g.start; i <= g.end; i++) {
      const num = String(i).padStart(3, '0');

      // <img> responsive
      const img = document.createElement('img');
      img.alt = `Gallery ${num}`;
      img.loading = 'lazy';
      img.dataset.cld = `gallery-${num}`;

      // <a> untuk lightbox/full-size
      const a = document.createElement('a');
      a.href = `https://res.cloudinary.com/${CLOUD_NAME}/image/upload/f_auto,q_auto/gallery-${num}.avif`;
      a.setAttribute('data', 'pop');
      a.appendChild(img);

      gallery.appendChild(a);
    }
  });

  function cloudinaryURL(publicId, width) {
    return `https://res.cloudinary.com/${CLOUD_NAME}/image/upload/f_auto,q_auto,w_${width}/${publicId}.avif`;
  }

  function updateImages() {
    document.querySelectorAll('img[data-cld]').forEach(img => {
      const colWidth = img.clientWidth || img.parentElement.clientWidth || 180;
      const dpr = window.devicePixelRatio || 1;
      const finalWidth = Math.min(Math.round(colWidth * dpr), MAX_WIDTH);
      const id = img.dataset.cld;
      img.src = cloudinaryURL(id, finalWidth);
    });
  }

  function debounce(fn, delay) {
    let t;
    return () => {
      clearTimeout(t);
      t = setTimeout(fn, delay);
    };
  }

  requestAnimationFrame(updateImages);
  window.addEventListener('resize', debounce(updateImages, 200));
})();
</script>
