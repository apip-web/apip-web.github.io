  display: block;
}

<script>
(function () {
  const CLOUD_NAME = 'dpawix5pp';
  const MAX_WIDTH = 1200;

  const galleries = [
    { selector: '.gallery-1', start: 1, end: 6 },
    { selector: '.gallery-2', start: 7, end: 12 }
  ];

  galleries.forEach(g => {
    const gallery = document.querySelector(g.selector);
    if (!gallery) return;

    for (let i = g.start; i <= g.end; i++) {
      const num = String(i).padStart(3, '0');
      const publicId = `gallery-${num}`;

      // 1. <img> Thumbnail (Kecil) - Paksa .avif
      const img = document.createElement('img');
      img.alt = `Gallery ${num}`;
      img.loading = 'lazy';
      img.dataset.cld = publicId;
      img.src = "data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7";

      // 2. <a> Gambar Besar (Original)
      // KUNCI: Tambahkan .jpg di akhir sebagai "hint" untuk browser
      // f_auto akan tetap mengubahnya jadi format terbaik (avif/webp) secara internal
      const a = document.createElement('a');
      a.href = `https://res.cloudinary.com/${CLOUD_NAME}/image/upload/f_auto,q_auto/${publicId}.jpg`;
      a.setAttribute('data', 'pop');
      a.appendChild(img);

      gallery.appendChild(a);
    }
  });

  // Fungsi untuk Thumbnail (Kecil) - Tetap .avif
  function cloudinaryURL(publicId, width) {
    return `https://res.cloudinary.com/${CLOUD_NAME}/image/upload/f_auto,q_auto,w_${width}/${publicId}.avif`;
  }

  function updateImages() {
    document.querySelectorAll('img[data-cld]').forEach(img => {
      const colWidth = img.parentElement?.clientWidth || 300;
      const dpr = window.devicePixelRatio || 1;
      const finalWidth = Math.min(Math.round(colWidth * dpr), MAX_WIDTH);
      
      const id = img.dataset.cld;
      const newSrc = cloudinaryURL(id, finalWidth);
      
      if (img.src !== newSrc) {
        img.src = newSrc;
      }
    });
  }

  setTimeout(updateImages, 150);
  
  // Sisanya tetap sama (debounce, resize listener, dll)
  // ...
})();
</script>
