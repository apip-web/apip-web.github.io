<h1 id="tag-title">All Posts</h1>

<div id="posts">
  {% for post in site.posts %}
    <article class="post" data-tags="{{ post.tags | join: ',' }}">
      <h2>
        <a href="{{ post.url }}">{{ post.title }}</a>
      </h2>

      <div class="post-tags">
        {% for tag in post.tags %}
          <button class="tag" data-tag="{{ tag }}">#{{ tag }}</button>
        {% endfor %}
      </div>
    </article>
  {% endfor %}
</div>

<script>
function filterByTag(tag) {
  const title = document.getElementById('tag-title');
  if (!title) return;

  document.querySelectorAll('.post').forEach(post => {
    const tags = post.dataset.tags ? post.dataset.tags.split(',') : [];
    post.style.display = (tag === '' || tags.includes(tag)) ? '' : 'none';
  });

  title.textContent = tag ? `Showing all posts under "${tag}"` : 'All Posts';
}

// Fungsi untuk membaca hash URL dan menjalankan filter
function handleHash() {
  const hash = decodeURIComponent(location.hash.replace('#', ''));
  filterByTag(hash);
}

// 1. Jalankan saat klik tombol tag di dalam halaman ini
document.addEventListener('click', function (e) {
  const btn = e.target.closest('.tag');
  if (!btn) return;
  
  // Jangan preventDefault agar location.hash berubah otomatis
  const tag = btn.dataset.tag;
  location.hash = tag; 
});

// 2. Tombol Show All
const showAllBtn = document.getElementById('show-all');
if (showAllBtn) {
  showAllBtn.addEventListener('click', () => {
    location.hash = ''; // Menghapus hash akan memicu hashchange
  });
}

// 3. LISTENERS KRUSIAL:
window.addEventListener('hashchange', handleHash); // Cek saat hash berubah
window.addEventListener('DOMContentLoaded', handleHash); // Cek saat load awal
handleHash(); // Jalankan langsung jika masuk via navigasi SPA
</script>

<button id="show-all">All</button>
